## Install

```bash
npm i --save lucide-ripple
#or
yarn add lucide-ripple
#or
pnpm add lucide-ripple
```

## Usage

```jsx
import { AArrowDown } from "lucide-ripple";

export component Page() {
  <AArrowDown />
  <BadgeDollarSign size={40} />
  <Apple color="blue" />
  <Barrel class="size-16 text-red-400" />
}
```
